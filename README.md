which supports Horizontal and Vertical.

 - android:orientation
 - point_progress
 - point_max
 - point_small_radius
 - point_block_radius
 - point_progress_height
 - point_progress_full_color
 - point_progress_empty_color
 - point_border_line_color
 - point_border_fill_color
 - point_border_line_width
